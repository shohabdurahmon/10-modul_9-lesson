class ResetPasswordView(TemplateView):
template_name = 'auth/reset-password.html'

def post(self, request, *args, **kwargs):
code = request.POST.get('code')
password = request.POST.get('password')
confirm_password = request.POST.get('confirm_password')

if code != str(request.session.get('reset_code')):Kiritilgan kod sessiyada saqlangan kod bilan mos bo‘lishi kerak.
return render(request, self.template_name, {'error': 'Kod noto‘g‘ri'})

if password != confirm_password:Ikkala parol bir xil bo‘lishi kerak.
return render(request, self.template_name, {'error': 'Parollar bir xil emas'})

user = User.objects.filter(id=request.session.get('reset_user_id')).first()
if not user:
return redirect('forgot-password')
